## Core Imports

### Aztec Node Client

```typescript
// Create a connection to Aztec node
import { createAztecNodeClient } from "@aztec/aztec.js/node";

const node = createAztecNodeClient("https://devnet-6.aztec-labs.com");
```

### PXE Client

```typescript
// For custom PXE connections
import { createPXEClient } from "@aztec/aztec.js";
```

### Aztec Address

```typescript
// Address type for accounts and contracts
import { AztecAddress } from "@aztec/stdlib/aztec-address";

const address = AztecAddress.fromString("0x1586f476...");
const zeroAddress = AztecAddress.ZERO;
```

## Wallet Imports

### Browser (Lazy Loading) - RECOMMENDED

```typescript
// For web apps - loads WASM on demand
const { TestWallet } = await import("@aztec/test-wallet/client/lazy");

const wallet = await TestWallet.create(node, { proverEnabled: true });
```

### Server/Node.js

```typescript
// For CLI scripts and server-side code
import { TestWallet } from "@aztec/test-wallet/server";

const wallet = await TestWallet.create(node, { proverEnabled: true });
```

### Creating Accounts

```typescript
// Create Schnorr account
const accountManager = await wallet.createSchnorrAccount(
  secretKey,  // Fr
  salt,       // Fr
  signingKey  // GrumpkinScalar
);

// Get account address
const accountAddress = accountManager.address;

// Get deployment method
const deployMethod = await accountManager.getDeployMethod();
```

## Cryptographic Types

### Field Element (Fr)

```typescript
// Finite field element - basic scalar type
import { Fr } from "@aztec/foundation/curves/bn254";

const randomValue = Fr.random();
const fromString = Fr.fromString("12345");
const zero = Fr.ZERO;
```

### Grumpkin Scalar

```typescript
// For signing keys
import { GrumpkinScalar } from "@aztec/foundation/curves/grumpkin";

const signingKey = GrumpkinScalar.random();
```

## Contract Imports

### SponsoredFPC Contract

```typescript
// For fee payment
import { SponsoredFPCContract } from "@aztec/noir-contracts.js/SponsoredFPC";

// Register contract
await wallet.registerContract(instance, SponsoredFPCContract.artifact);
```

### Generated Contract Artifacts

```typescript
// Import your generated contract
import { CounterContract } from "../artifacts/Counter";

// Deploy
const contract = await CounterContract.deploy(wallet)
  .send({ from: accountAddress, fee: { paymentMethod } })
  .deployed();

// Call methods
const count = await contract.methods.get_count().simulate({ from: accountAddress });
```

## Fee Handling

### Sponsored Fee Payment Method

```typescript
import { SponsoredFeePaymentMethod } from "@aztec/aztec.js/fee";

const sponsoredFpcAddress = AztecAddress.fromString(SPONSORED_FPC_ADDRESS);
const paymentMethod = new SponsoredFeePaymentMethod(sponsoredFpcAddress);

// Use in transactions
await contract.methods.increment()
  .send({ 
    from: accountAddress, 
    fee: { paymentMethod } 
  })
  .wait();
```

## Complete Import Example

```typescript
/**
 * Complete Aztec web app imports
 */

// Core Aztec.js imports
import { createAztecNodeClient } from "@aztec/aztec.js/node";
import { SponsoredFeePaymentMethod } from "@aztec/aztec.js/fee";

// Standard library imports
import { AztecAddress } from "@aztec/stdlib/aztec-address";

// Cryptographic types
import { Fr } from "@aztec/foundation/curves/bn254";
import { GrumpkinScalar } from "@aztec/foundation/curves/grumpkin";

// Pre-deployed contracts
import { SponsoredFPCContract } from "@aztec/noir-contracts.js/SponsoredFPC";

// Your contract (generated by aztec codegen)
import { CounterContract } from "../artifacts/Counter";

// Wallet - lazy loaded for browser
const { TestWallet } = await import("@aztec/test-wallet/client/lazy");

// Constants
const DEVNET_NODE_URL = "https://devnet-6.aztec-labs.com";
const SPONSORED_FPC_ADDRESS = "0x1586f476995be97f07ebd415340a14be48dc28c6c661cc6bdddb80ae790caa4e";

// Initialize
async function init() {
  const node = createAztecNodeClient(DEVNET_NODE_URL);
  const wallet = await TestWallet.create(node, { proverEnabled: true });
  
  // Create account
  const accountManager = await wallet.createSchnorrAccount(
    Fr.random(),
    Fr.random(),
    GrumpkinScalar.random()
  );
  
  // Register SponsoredFPC
  const sponsoredFpcAddress = AztecAddress.fromString(SPONSORED_FPC_ADDRESS);
  const sponsoredFpcInstance = await node.getContract(sponsoredFpcAddress);
  await wallet.registerContract(sponsoredFpcInstance, SponsoredFPCContract.artifact);
  
  return { node, wallet, accountManager };
}
```

## Import Patterns Summary

| What | Import Path | Use Case |
|------|-------------|----------|
| `createAztecNodeClient` | `@aztec/aztec.js/node` | Connect to node |
| `TestWallet` (browser) | `@aztec/test-wallet/client/lazy` | Browser wallet |
| `TestWallet` (server) | `@aztec/test-wallet/server` | Node.js wallet |
| `AztecAddress` | `@aztec/stdlib/aztec-address` | Addresses |
| `Fr` | `@aztec/foundation/curves/bn254` | Field elements |
| `GrumpkinScalar` | `@aztec/foundation/curves/grumpkin` | Signing keys |
| `SponsoredFeePaymentMethod` | `@aztec/aztec.js/fee` | Fee payment |
| `SponsoredFPCContract` | `@aztec/noir-contracts.js/SponsoredFPC` | Fee contract |
